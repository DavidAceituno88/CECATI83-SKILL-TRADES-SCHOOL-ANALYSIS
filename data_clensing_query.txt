SELECT [ciclo_ins] AS Period
      ,[fecha_ini] date
      ,[per_egre]
      ,[docente] AS Teacher
      ,[nom_est] AS State
	  ,CONCAT(nombre,' ',a_paterno,' ',a_materno) AS Student_Name
      ,[colonia] AS Street
      ,[muni_deleg] AS City
      ,[cp] AS Postal Code
      ,[sexo] AS Gender
      ,[edad] AS Age
      ,[e_civil] AS Civil_Status
      ,[telefono] AS Phone
      ,[email] 
      ,[f_ingreso] AS Start_Date
      ,[escolarida] AS Scolarship
      ,[pais] AS Country
      ,[nom_cur] AS Course
      ,[tipo_med] AS Means
      ,[tipo_mot] AS Motive
      ,[trabaja_si] 
      ,[trabaja_no]
	  ,CASE WHEN trabaja_si = 1 THEN 'YES'
	  WHEN trabaja_no = 1 THEN 'NO'
	  ELSE 'NO'
	  END  AS HAS_A_JOB
      ,[ingresos_m] AS Monthly_Incom
      ,[f_inscrip] AS Registration_Date
      ,[nom_area] AS Area
      ,[nom_esp] AS Specialty
      ,[escol] 
      ,[turno] AS Shift
  FROM [olympic_games].[dbo].[CECATI_FULL]
  ORDER BY f_ingreso;
