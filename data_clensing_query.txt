SELECT [ciclo_ins]
      ,[fecha_ini]
      ,[per_egre]
      ,[docente] AS Docente
      ,[nom_est] AS Estado
	  ,CONCAT(nombre,' ',a_paterno,' ',a_materno) AS Nombre_Estudiante
      ,[colonia] AS Colonia
      ,[muni_deleg] AS Municipio
      ,[cp] 
      ,[sexo] AS Sexo
      ,[edad] AS Edad
      ,[e_civil] AS Estado_Civil
      ,[telefono] AS Telefono
      ,[email] 
      ,[f_ingreso] AS Fecha_Ingreso
      ,[escolarida] AS Escolaridad
      ,[pais] AS Pais
      ,[nom_cur] AS Curso
      ,[tipo_med] AS Medio
      ,[tipo_mot] AS Motivo
      ,[trabaja_si]
      ,[trabaja_no]
	  ,CASE WHEN trabaja_si = 1 THEN 'SI'
	  WHEN trabaja_no = 1 THEN 'NO'
	  ELSE 'NO'
	  END  AS Trabaja
      ,[ingresos_m] AS Ingresos_Mensuales
      ,[f_inscrip] AS Fecha_Inscripcion
      ,[nom_area] AS Area
      ,[nom_esp] AS Especialidad
      ,[escol] AS Escolaridad
      ,[turno] AS Turno
  FROM [olympic_games].[dbo].[CECATI_FULL]
  ORDER BY f_ingreso;
